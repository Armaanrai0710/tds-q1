import os
import io
import base64
import pandas as pd
import matplotlib.pyplot as plt

sample_csv = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,Sales,Europe,90.1,12,3.8
EMP002,Marketing,Middle East,71.27,8,3.7
EMP003,Marketing,Latin America,61.17,2,4.2
EMP004,Marketing,North America,69.67,12,3.6
EMP005,Marketing,Asia Pacific,86,1,3.3
"""

csv_path = "employees.csv"
if os.path.exists(csv_path):
    df = pd.read_csv(csv_path)
else:
    df = pd.read_csv(io.StringIO(sample_csv))

marketing_count = (df["department"].astype(str) == "Marketing").sum()
print("Marketing frequency count:", marketing_count)

dept_counts = df["department"].astype(str).value_counts().sort_values(ascending=False)

plt.figure(figsize=(8, 6))
dept_counts.plot(kind="bar")
plt.title("Department Distribution (Employee Count)")
plt.xlabel("Department")
plt.ylabel("Count")
plt.tight_layout()
plt.savefig("department_hist.png", dpi=120)
